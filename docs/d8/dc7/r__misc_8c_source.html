324 ===============
325 R_ViewChanged
326 
327 Called every time the vid structure or r_refdef changes.
328 Guaranteed to be called before the first refresh
329 ===============
330 */
331 void R_ViewChanged (vrect_t *vr)
332 {
333  int i;
334 
335  r_refdef.vrect = *vr;
336 
337  r_refdef.horizontalFieldOfView = 2*tan((float)r_newrefdef.fov_x/360*M_PI);
338  verticalFieldOfView = 2*tan((float)r_newrefdef.fov_y/360*M_PI);
339 
340  r_refdef.fvrectx = (float)r_refdef.vrect.x;
341  r_refdef.fvrectx_adj = (float)r_refdef.vrect.x - 0.5;
342  r_refdef.vrect_x_adj_shift20 = (r_refdef.vrect.x<<20) + (1<<19) - 1;
343  r_refdef.fvrecty = (float)r_refdef.vrect.y;
344  r_refdef.fvrecty_adj = (float)r_refdef.vrect.y - 0.5;
345  r_refdef.vrectright = r_refdef.vrect.x + r_refdef.vrect.width;
346  r_refdef.vrectright_adj_shift20 = (r_refdef.vrectright<<20) + (1<<19) - 1;
347  r_refdef.fvrectright = (float)r_refdef.vrectright;
348  r_refdef.fvrectright_adj = (float)r_refdef.vrectright - 0.5;
349  r_refdef.vrectrightedge = (float)r_refdef.vrectright - 0.99;
350  r_refdef.vrectbottom = r_refdef.vrect.y + r_refdef.vrect.height;
351  r_refdef.fvrectbottom = (float)r_refdef.vrectbottom;
352  r_refdef.fvrectbottom_adj = (float)r_refdef.vrectbottom - 0.5;
353 
354  r_refdef.aliasvrect.x = (int)(r_refdef.vrect.x * r_aliasuvscale);
355  r_refdef.aliasvrect.y = (int)(r_refdef.vrect.y * r_aliasuvscale);
356  r_refdef.aliasvrect.width = (int)(r_refdef.vrect.width * r_aliasuvscale);
357  r_refdef.aliasvrect.height = (int)(r_refdef.vrect.height * r_aliasuvscale);
358  r_refdef.aliasvrectright = r_refdef.aliasvrect.x +
359  r_refdef.aliasvrect.width;
360  r_refdef.aliasvrectbottom = r_refdef.aliasvrect.y +
361  r_refdef.aliasvrect.height;
362 
363  xOrigin = r_refdef.xOrigin;
364  yOrigin = r_refdef.yOrigin;
365 
366 // values for perspective projection
367 // if math were exact, the values would range from 0.5 to to range+0.5
368 // hopefully they wll be in the 0.000001 to range+.999999 and truncate
369 // the polygon rasterization will never render in the first row or column
370 // but will definately render in the [range] row and column, so adjust the
371 // buffer origin to get an exact edge to edge fill
372  xcenter = ((float)r_refdef.vrect.width * XCENTERING) +
373  r_refdef.vrect.x - 0.5;
374  aliasxcenter = xcenter * r_aliasuvscale;
375  ycenter = ((float)r_refdef.vrect.height * YCENTERING) +
376  r_refdef.vrect.y - 0.5;
377  aliasycenter = ycenter * r_aliasuvscale;
378 
379  xscale = r_refdef.vrect.width / r_refdef.horizontalFieldOfView;
380  aliasxscale = xscale * r_aliasuvscale;
381  xscaleinv = 1.0 / xscale;
382 
383  yscale = xscale;
384  aliasyscale = yscale * r_aliasuvscale;
385  yscaleinv = 1.0 / yscale;
386  xscaleshrink = (r_refdef.vrect.width-6)/r_refdef.horizontalFieldOfView;
387  yscaleshrink = xscaleshrink;
388 
389 // left side clip
390  screenedge[0].normal[0] = -1.0 / (xOrigin*r_refdef.horizontalFieldOfView);
391  screenedge[0].normal[1] = 0;
392  screenedge[0].normal[2] = 1;
393  screenedge[0].type = PLANE_ANYZ;
394 
395 // right side clip
396  screenedge[1].normal[0] =
397  1.0 / ((1.0-xOrigin)*r_refdef.horizontalFieldOfView);
398  screenedge[1].normal[1] = 0;
399  screenedge[1].normal[2] = 1;
400  screenedge[1].type = PLANE_ANYZ;
401 
402 // top side clip
403  screenedge[2].normal[0] = 0;
404  screenedge[2].normal[1] = -1.0 / (yOrigin*verticalFieldOfView);
405  screenedge[2].normal[2] = 1;
406  screenedge[2].type = PLANE_ANYZ;
407 
408 // bottom side clip
409  screenedge[3].normal[0] = 0;
410  screenedge[3].normal[1] = 1.0 / ((1.0-yOrigin)*verticalFieldOfView);
411  screenedge[3].normal[2] = 1;
412  screenedge[3].type = PLANE_ANYZ;
413 
414  for (i=0 ; i<4 ; i++)
415  VectorNormalize (screenedge[i].normal);
416 
417  D_ViewChanged ();
418 }

549 /*
550 ==============
551 WritePCXfile
552 ==============
553 */
554 void WritePCXfile (char *filename, byte *data, int width, int height,
555  int rowbytes, byte *palette)
556 {
557  int i, j, length;
558  pcx_t *pcx;
559  byte *pack;
560  FILE *f;
561 
562  pcx = (pcx_t *)malloc (width*height*2+1000);
563  if (!pcx)
564  return;
565 
566  pcx->manufacturer = 0x0a; // PCX id
567  pcx->version = 5; // 256 color
568  pcx->encoding = 1; // uncompressed
569  pcx->bits_per_pixel = 8; // 256 color
570  pcx->xmin = 0;
571  pcx->ymin = 0;
572  pcx->xmax = LittleShort((short)(width-1));
573  pcx->ymax = LittleShort((short)(height-1));
574  pcx->hres = LittleShort((short)width);
575  pcx->vres = LittleShort((short)height);
576  memset (pcx->palette,0,sizeof(pcx->palette));
577  pcx->color_planes = 1; // chunky image
578  pcx->bytes_per_line = LittleShort((short)width);
579  pcx->palette_type = LittleShort(2); // not a grey scale
580  memset (pcx->filler,0,sizeof(pcx->filler));
581 
582 // pack the image
583  pack = &pcx->data;
584 
585  for (i=0 ; i<height ; i++)
586  {
587  for (j=0 ; j<width ; j++)
588  {
589  if ( (*data & 0xc0) != 0xc0)
590  *pack++ = *data++;
591  else
592  {
593  *pack++ = 0xc1;
594  *pack++ = *data++;
595  }
596  }
597 
598  data += rowbytes - width;
599  }
600 
601 // write the palette
602  *pack++ = 0x0c; // palette ID byte
603  for (i=0 ; i<768 ; i++)
604  *pack++ = *palette++;
605 
606 // write output file
607  length = pack - (byte *)pcx;
608  f = fopen (filename, "wb");
609  if (!f)
610  ri.Con_Printf (PRINT_ALL, "Failed to open to %s\n", filename);
611  else
612  {
613  fwrite ((void *)pcx, 1, length, f);
614  fclose (f);
615  }
616 
617  free (pcx);
618 }
