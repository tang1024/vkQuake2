47 /*
48 ================
49 Draw_Char
50 
51 Draws one 8*8 graphics character with 0 being transparent.
52 It can be clipped to the top of the screen to allow the console to be
53 smoothly scrolled off.
54 ================
55 */
56 void Draw_Char (int x, int y, int num)
57 {
58  int row, col;
59  float frow, fcol, size;
60 
61  num &= 255;
62 
63  if ( (num&127) == 32 )
64  return; // space
65 
66  if (y <= -8)
67  return; // totally off screen
68 
69  cvar_t *scale = ri.Cvar_Get("hudscale", "1", 0);
70 
71  row = num>>4;
72  col = num&15;
73 
74  frow = row*0.0625;
75  fcol = col*0.0625;
76  size = 0.0625;
77 
78  GL_Bind (draw_chars->texnum);
79 
80  qglBegin (GL_QUADS);
81  qglTexCoord2f (fcol, frow);
82  qglVertex2f (x, y);
83  qglTexCoord2f (fcol + size, frow);
84  qglVertex2f (x+8*scale->value, y);
85  qglTexCoord2f (fcol + size, frow + size);
86  qglVertex2f (x+8*scale->value, y+8*scale->value);
87  qglTexCoord2f (fcol, frow + size);
88  qglVertex2f (x, y+8*scale->value);
89  qglEnd ();
90 }

325 void Draw_StretchRaw (int x, int y, int w, int h, int cols, int rows, byte *data)

335

337 
338  if (rows<=256)
339  {
340  hscale = 1;
341  trows = rows;
342  }
343  else
344  {
345  hscale = rows/256.0;
346  trows = 256;
347  }
348  t = rows*hscale / 256;
349 
350  if ( !qglColorTableEXT )

352  unsigned *dest;
353

356  row = (int)(i*hscale);

358  break;
359  source = data + cols*row;

362  frac = fracstep >> 1;
363  for (j=0 ; j<256 ; j++)
364  {

366  frac += fracstep;
367  }
368  }
369

371  }
