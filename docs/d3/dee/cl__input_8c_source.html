31 /*
32 ===============================================================================
33 
34 KEY BUTTONS
35 
36 Continuous button event tracking is complicated by the fact that two different
37 input sources (say, mouse button 1 and the control key) can both press the
38 same button, but the button should only be released when both of the
39 pressing key have been released.
40 
41 When a key event issues a button command (+forward, +attack, etc), it appends
42 its key number as a parameter to the command so it can be matched up with
43 the release.
44 
45 state bit 0 is the current state of the key
46 state bit 1 is edge triggered on the up to down transition
47 state bit 2 is edge triggered on the down to up transition
48 
49 
50 Key_Event (int key, qboolean down, unsigned time);
51 
52  +mlook src time
53 
54 ===============================================================================
55 */

67 void KeyDown (kbutton_t *b)
68 {
69  int k;
70  char *c;
71 
72  c = Cmd_Argv(1);
73  if (c[0])
74  k = atoi(c);
75  else
76  k = -1; // typed manually at the console for continuous down
77 
78  if (k == b->down[0] || k == b->down[1])
79  return; // repeating key
80 
81  if (!b->down[0])
82  b->down[0] = k;
83  else if (!b->down[1])
84  b->down[1] = k;
85  else
86  {
87  Com_Printf ("Three keys down for a button!\n");
88  return;
89  }
90 
91  if (b->state & 1)
92  return; // still down
93 
94  // save timestamp
95  c = Cmd_Argv(2);
96  b->downtime = atoi(c);
97  if (!b->downtime)
98  b->downtime = sys_frame_time - 100;
99 
100  b->state |= 1 + 2; // down + impulse down
101 }

103 void KeyUp (kbutton_t *b)
104 {
105  int k;
106  char *c;
107  unsigned uptime;
108 
109  c = Cmd_Argv(1);
110  if (c[0])
111  k = atoi(c);
112  else
113  { // typed manually at the console, assume for unsticking, so clear all
114  b->down[0] = b->down[1] = 0;
115  b->state = 4; // impulse up
116  return;
117  }
118 
119  if (b->down[0] == k)
120  b->down[0] = 0;
121  else if (b->down[1] == k)
122  b->down[1] = 0;
123  else
124  return; // key up without coresponding down (menu pass through)
125  if (b->down[0] || b->down[1])
126  return; // some other key is still holding it down
127 
128  if (!(b->state & 1))
129  return; // still up (this should not happen)
130 
131  // save timestamp
132  c = Cmd_Argv(2);
133  uptime = atoi(c);
134  if (uptime)
135  b->msec += uptime - b->downtime;
136  else
137  b->msec += 10;
138 
139  b->state &= ~1; // now up
140  b->state |= 4; // impulse up
141 }

179 /*
180 ===============
181 CL_KeyState
182 
183 Returns the fraction of the frame that the key was down
184 ===============
185 */
186 float CL_KeyState (kbutton_t *key)
187 {
188  float val;
189  int msec;
190 
191  key->state &= 1; // clear impulses
192 
193  msec = key->msec;
194  key->msec = 0;
195 
196  if (key->state)
197  { // still down
198  msec += sys_frame_time - key->downtime;
199  key->downtime = sys_frame_time;
200  }
201 
202 #if 0
203  if (msec)
204  {
205  Com_Printf ("%i ", msec);
206  }
207 #endif
208 
209  val = (float)msec / frame_msec;
210  if (val < 0)
211  val = 0;
212  if (val > 1)
213  val = 1;
214 
215  return val;
216 }

235 /*
236 ================
237 CL_AdjustAngles
238 
239 Moves the local angle positions
240 ================
241 */
242 void CL_AdjustAngles (void)
243 {
244  float speed;
245  float up, down;
246 
247  if (in_speed.state & 1)
248  speed = cls.frametime * cl_anglespeedkey->value;
249  else
250  speed = cls.frametime;
251 
252  if (!(in_strafe.state & 1))
253  {
254  cl.viewangles[YAW] -= speed*cl_yawspeed->value*CL_KeyState (&in_right);
255  cl.viewangles[YAW] += speed*cl_yawspeed->value*CL_KeyState (&in_left);
256  }
257  if (in_klook.state & 1)
258  {
259  cl.viewangles[PITCH] -= speed*cl_pitchspeed->value * CL_KeyState (&in_forward);
260  cl.viewangles[PITCH] += speed*cl_pitchspeed->value * CL_KeyState (&in_back);
261  }
262 
263  up = CL_KeyState (&in_lookup);
264  down = CL_KeyState(&in_lookdown);
265 
266  cl.viewangles[PITCH] -= speed*cl_pitchspeed->value * up;
267  cl.viewangles[PITCH] += speed*cl_pitchspeed->value * down;
268 }

270 /*
271 ================
272 CL_BaseMove
273 
274 Send the intended movement message to the server
275 ================
276 */
277 void CL_BaseMove (usercmd_t *cmd)
278 {
279  CL_AdjustAngles ();
280 
281  memset (cmd, 0, sizeof(*cmd));
282 
283  VectorCopy (cl.viewangles, cmd->angles);
284  if (in_strafe.state & 1)
285  {
286  cmd->sidemove += cl_sidespeed->value * CL_KeyState (&in_right);
287  cmd->sidemove -= cl_sidespeed->value * CL_KeyState (&in_left);
288  }
289 
290  cmd->sidemove += cl_sidespeed->value * CL_KeyState (&in_moveright);
291  cmd->sidemove -= cl_sidespeed->value * CL_KeyState (&in_moveleft);
292 
293  cmd->upmove += cl_upspeed->value * CL_KeyState (&in_up);
294  cmd->upmove -= cl_upspeed->value * CL_KeyState (&in_down);
295 
296  if (! (in_klook.state & 1) )
297  {
298  cmd->forwardmove += cl_forwardspeed->value * CL_KeyState (&in_forward);
299  cmd->forwardmove -= cl_forwardspeed->value * CL_KeyState (&in_back);
300  }
301 
302 //
303 // adjust for speed key / running
304 //
305  if ( (in_speed.state & 1) ^ (int)(cl_run->value) )
306  {
307  cmd->forwardmove *= 2;
308  cmd->sidemove *= 2;
309  cmd->upmove *= 2;
310  }
311 }

313 void CL_ClampPitch (void)
314 {
315  float pitch;
316 
317  pitch = SHORT2ANGLE(cl.frame.playerstate.pmove.delta_angles[PITCH]);
318  if (pitch > 180)
319  pitch -= 360;
320 
321  if (cl.viewangles[PITCH] + pitch < -360)
322  cl.viewangles[PITCH] += 360; // wrapped
323  if (cl.viewangles[PITCH] + pitch > 360)
324  cl.viewangles[PITCH] -= 360; // wrapped
325 
326  if (cl.viewangles[PITCH] + pitch > 89)
327  cl.viewangles[PITCH] = 89 - pitch;
328  if (cl.viewangles[PITCH] + pitch < -89)
329  cl.viewangles[PITCH] = -89 - pitch;
330 }

332 /*
333 ==============
334 CL_FinishMove
335 ==============
336 */
337 void CL_FinishMove (usercmd_t *cmd)
338 {
339  int ms;
340  int i;
341 
342 //
343 // figure button bits
344 //
345  if ( in_attack.state & 3 )
346  cmd->buttons |= BUTTON_ATTACK;
347  in_attack.state &= ~2;
348 
349  if (in_use.state & 3)
350  cmd->buttons |= BUTTON_USE;
351  in_use.state &= ~2;
352 
353  if (anykeydown && cls.key_dest == key_game)
354  cmd->buttons |= BUTTON_ANY;
355 
356  // send milliseconds of time to apply the move
357  ms = cls.frametime * 1000;
358  if (ms > 250)
359  ms = 100; // time was unreasonable
360  cmd->msec = ms;
361 
362  CL_ClampPitch ();
363  for (i=0 ; i<3 ; i++)
364  cmd->angles[i] = ANGLE2SHORT(cl.viewangles[i]);
365 
366  cmd->impulse = in_impulse;
367  in_impulse = 0;
368 
369 // send the ambient light level at the player's current position
370  cmd->lightlevel = (byte)cl_lightlevel->value;
371 }

378 usercmd_t CL_CreateCmd (void)
379 {
380  usercmd_t cmd;
381 
382  frame_msec = sys_frame_time - old_sys_frame_time;
383  if (frame_msec < 1)
384  frame_msec = 1;
385  if (frame_msec > 200)
386  frame_msec = 200;
387 
388  // get basic movement from keyboard
389  CL_BaseMove (&cmd);
390 
391  // allow mice or other external controllers to add to the move
392  IN_Move (&cmd);
393 
394  CL_FinishMove (&cmd);
395 
396  old_sys_frame_time = sys_frame_time;
397 
398 //cmd.impulse = cls.framecount;
399 
400  return cmd;
401 }

456 =================
457 CL_SendCmd
458 =================
459 */
460 void CL_SendCmd (void)
461 {
462  sizebuf_t buf;
463  byte data[128];
464  int i;
465  usercmd_t *cmd, *oldcmd;
466  usercmd_t nullcmd;
467  int checksumIndex;
468 
469  memset( &buf, 0, sizeof(buf) );
470 
471  // build a command even if not connected
472 
473  // save this command off for prediction
474  i = cls.netchan.outgoing_sequence & (CMD_BACKUP-1);
475  cmd = &cl.cmds[i];
476  cl.cmd_time[i] = cls.realtime; // for netgraph ping calculation
477 
478  *cmd = CL_CreateCmd ();
479 
480  cl.cmd = *cmd;
481 
482  if (cls.state == ca_disconnected || cls.state == ca_connecting)
483  return;
484 
485  if ( cls.state == ca_connected)
486  {
487  if (cls.netchan.message.cursize || curtime - cls.netchan.last_sent > 1000 )
488  Netchan_Transmit (&cls.netchan, 0, buf.data);
489  return;
490  }
491 
492  // send a userinfo update if needed
493  if (userinfo_modified)
494  {
495  CL_FixUpGender();
496  userinfo_modified = false;
497  MSG_WriteByte (&cls.netchan.message, clc_userinfo);
498  MSG_WriteString (&cls.netchan.message, Cvar_Userinfo() );
499  }
500 
501  SZ_Init (&buf, data, sizeof(data));
502 
503  if (cmd->buttons && cl.cinematictime > 0 && !cl.attractloop
504  && cls.realtime - cl.cinematictime > 1000)
505  { // skip the rest of the cinematic
506  SCR_FinishCinematic ();
507  }
508 
509  // begin a client move command
510  MSG_WriteByte (&buf, clc_move);
511 
512  // save the position for a checksum byte
513  checksumIndex = buf.cursize;
514  MSG_WriteByte (&buf, 0);
515 
516  // let the server know what the last frame we
517  // got was, so the next message can be delta compressed
518  if (cl_nodelta->value || !cl.frame.valid || cls.demowaiting)
519  MSG_WriteLong (&buf, -1); // no compression
520  else
521  MSG_WriteLong (&buf, cl.frame.serverframe);
522 
523  // send this and the previous cmds in the message, so
524  // if the last packet was dropped, it can be recovered
525  i = (cls.netchan.outgoing_sequence-2) & (CMD_BACKUP-1);
526  cmd = &cl.cmds[i];
527  memset (&nullcmd, 0, sizeof(nullcmd));
528  MSG_WriteDeltaUsercmd (&buf, &nullcmd, cmd);
529  oldcmd = cmd;
530 
531  i = (cls.netchan.outgoing_sequence-1) & (CMD_BACKUP-1);
532  cmd = &cl.cmds[i];
533  MSG_WriteDeltaUsercmd (&buf, oldcmd, cmd);
534  oldcmd = cmd;
535 
536  i = (cls.netchan.outgoing_sequence) & (CMD_BACKUP-1);
537  cmd = &cl.cmds[i];
538  MSG_WriteDeltaUsercmd (&buf, oldcmd, cmd);
539 
540  // calculate a checksum over the move commands
541  buf.data[checksumIndex] = COM_BlockSequenceCRCByte(
542  buf.data + checksumIndex + 1, buf.cursize - checksumIndex - 1,
543  cls.netchan.outgoing_sequence);
544 
545  //
546  // deliver the message
547  //
548  Netchan_Transmit (&cls.netchan, buf.cursize, buf.data);
549 }
